CREATE TABLE `booking` (
	`ID` INT(11) NOT NULL AUTO_INCREMENT,
	`Name` VARCHAR(50) NULL DEFAULT NULL,
	`Phone` VARCHAR(50) NULL DEFAULT NULL,
	`StreetNum` VARCHAR(50) NULL DEFAULT NULL,
	`StreetName` VARCHAR(50) NULL DEFAULT NULL,
	`Suburb` VARCHAR(50) NULL DEFAULT NULL,
	`DestSurb` VARCHAR(50) NULL DEFAULT NULL,
	`DateTime` VARCHAR(50) NULL DEFAULT NULL,
	`BookDate` VARCHAR(50) NULL DEFAULT NULL,
	`Assign` VARCHAR(50) NULL DEFAULT 'Unassign',
	PRIMARY KEY (`ID`)
)
